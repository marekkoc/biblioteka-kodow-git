Ok, stwórzmy GUI używając PyQt5. Oto kompletny kod:

```python
import sys
from PyQt5.QtWidgets import (QApplication, QMainWindow, QPushButton, QLabel, 
                           QVBoxLayout, QHBoxLayout, QWidget, QFileDialog,
                           QTextEdit, QScrollArea, QStatusBar, QGroupBox)
from PyQt5.QtCore import Qt
from PyQt5.QtGui import QPixmap, QImage
import cv2
import numpy as np
from pathlib import Path
from product_processor import ProductImageProcessor  # nasz poprzedni kod

class ProductAnalyzerGUI(QMainWindow):
    def __init__(self):
        super().__init__()
        self.processor = ProductImageProcessor()
        self.current_image_path = None
        self.current_processed_image = None
        self.initUI()

    def initUI(self):
        """Inicjalizacja interfejsu użytkownika"""
        self.setWindowTitle('Product Nutrition Analyzer')
        self.setGeometry(100, 100, 1400, 800)

        # Główny widget i layout
        main_widget = QWidget()
        self.setCentralWidget(main_widget)
        main_layout = QHBoxLayout(main_widget)

        # Lewa strona - panel z obrazami
        left_panel = QGroupBox("Images")
        left_layout = QVBoxLayout()
        
        # Przyciski akcji
        button_layout = QHBoxLayout()
        self.load_btn = QPushButton('Load Image')
        self.process_btn = QPushButton('Process Image')
        self.save_btn = QPushButton('Save Results')
        self.process_btn.setEnabled(False)
        self.save_btn.setEnabled(False)
        
        button_layout.addWidget(self.load_btn)
        button_layout.addWidget(self.process_btn)
        button_layout.addWidget(self.save_btn)
        
        # Podgląd obrazów
        self.original_image = QLabel()
        self.original_image.setMinimumSize(400, 300)
        self.original_image.setAlignment(Qt.AlignCenter)
        self.original_image.setStyleSheet("border: 1px solid black")
        
        self.processed_image = QLabel()
        self.processed_image.setMinimumSize(400, 300)
        self.processed_image.setAlignment(Qt.AlignCenter)
        self.processed_image.setStyleSheet("border: 1px solid black")

        left_layout.addLayout(button_layout)
        left_layout.addWidget(QLabel("Original Image:"))
        left_layout.addWidget(self.original_image)
        left_layout.addWidget(QLabel("Processed Image:"))
        left_layout.addWidget(self.processed_image)
        left_panel.setLayout(left_layout)

        # Prawa strona - wyniki
        right_panel = QGroupBox("Results")
        right_layout = QVBoxLayout()
        
        # Pole tekstowe na wyniki JSON
        self.result_text = QTextEdit()
        self.result_text.setReadOnly(True)
        right_layout.addWidget(self.result_text)
        right_panel.setLayout(right_layout)

        # Dodanie paneli do głównego layoutu
        main_layout.addWidget(left_panel, 1)
        main_layout.addWidget(right_panel, 1)

        # Pasek statusu
        self.statusBar = QStatusBar()
        self.setStatusBar(self.statusBar)

        # Podłączenie sygnałów
        self.load_btn.clicked.connect(self.load_image)
        self.process_btn.clicked.connect(self.process_image)
        self.save_btn.clicked.connect(self.save_results)

    def load_image(self):
        """Wczytanie obrazu z pliku"""
        file_name, _ = QFileDialog.getOpenFileName(
            self, 
            "Select Image", 
            "", 
            "Image Files (*.png *.jpg *.jpeg *.bmp)"
        )
        
        if file_name:
            self.current_image_path = file_name
            pixmap = QPixmap(file_name)
            scaled_pixmap = pixmap.scaled(
                self.original_image.size(), 
                Qt.KeepAspectRatio, 
                Qt.SmoothTransformation
            )
            self.original_image.setPixmap(scaled_pixmap)
            self.process_btn.setEnabled(True)
            self.statusBar.showMessage(f'Loaded: {file_name}')

    def process_image(self):
        """Przetwarzanie obrazu"""
        if self.current_image_path:
            try:
                # Przetwarzanie obrazu
                processed_data = self.processor.process_image(self.current_image_path)
                
                # Wyświetlenie przetworzonego obrazu
                processed_img = self.processor.last_processed_image
                if processed_img is not None:
                    height, width = processed_img.shape[:2]
                    bytes_per_line = 3 * width
                    q_img = QImage(
                        processed_img.data, 
                        width, 
                        height, 
                        bytes_per_line, 
                        QImage.Format_RGB888
                    )
                    pixmap = QPixmap.fromImage(q_img)
                    scaled_pixmap = pixmap.scaled(
                        self.processed_image.size(), 
                        Qt.KeepAspectRatio, 
                        Qt.SmoothTransformation
                    )
                    self.processed_image.setPixmap(scaled_pixmap)

                # Wyświetlenie wyników w formacie JSON
                import json
                json_str = json.dumps(processed_data, indent=4)
                self.result_text.setText(json_str)
                
                self.save_btn.setEnabled(True)
                self.statusBar.showMessage('Processing completed')
                
            except Exception as e:
                self.statusBar.showMessage(f'Error during processing: {str(e)}')
                
    def save_results(self):
        """Zapisanie wyników do pliku"""
        file_name, _ = QFileDialog.getSaveFileName(
            self, 
            "Save Results", 
            "", 
            "JSON Files (*.json)"
        )
        
        if file_name:
            try:
                with open(file_name, 'w', encoding='utf-8') as f:
                    f.write(self.result_text.toPlainText())
                self.statusBar.showMessage(f'Results saved to: {file_name}')
            except Exception as e:
                self.statusBar.showMessage(f'Error saving results: {str(e)}')

def main():
    app = QApplication(sys.argv)
    ex = ProductAnalyzerGUI()
    ex.show()
    sys.exit(app.exec_())

if __name__ == '__main__':
    main()
```

Ten kod tworzy aplikację z następującymi funkcjami:

1. Panel lewy:
   - Przycisk do wczytywania zdjęć
   - Wyświetlanie oryginalnego zdjęcia
   - Wyświetlanie przetworzonego zdjęcia
   - Przycisk do przetwarzania
   - Przycisk do zapisywania wyników

2. Panel prawy:
   - Pole tekstowe pokazujące wyniki w formacie JSON
   - Możliwość kopiowania i zapisywania wyników

3. Dodatkowe funkcje:
   - Pasek statusu pokazujący aktualny stan operacji
   - Skalowanie obrazów do rozmiaru okna
   - Obsługa błędów
   - Zapisywanie wyników do pliku
